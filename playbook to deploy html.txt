Install apache through playbook , deploy sample html file and show the output in browser using ip


- hosts: 172.31.3.3
  become: yes
  tasks:
   - name: install apache
     apt:
      name: apache2
      state: latest
   - name: start apache
     service:
      name: apache2
      state: started
   - name: copyfile
     copy:
      content: "hi nikhil"
      dest: /var/www/html/index.html
   - name: restart
     service:
      name: apache2
      state: restarted
